import pandas as pd

from tlo.lm import LinearModel, LinearModelType, Predictor

intercept = 2.0

df_ = pd.DataFrame(
    data={'prop1': [True, True, False], 'prop2': [2, 4, 6]}
)

lm = LinearModel(
    LinearModelType.ADDITIVE,
    intercept,
    Predictor('prop1').when(True, 1).otherwise(2),
    Predictor('prop2').when('>3', 3.0).otherwise(4.0),
)



[docs]def custom_lm(self, df, rng=None, **externals):
    result = pd.Series(intercept, index=df.index)
    result[df.prop1] += 1
    result[~df.prop1] += 2
    result[df.prop2 > 3] += 3.0
    result[df.prop2 <= 3] += 4.0
    return result



lm2 = LinearModel.custom(custom_lm)
